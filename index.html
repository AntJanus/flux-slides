var app = angular.module('app', []);

//factory fetches model data
app.factory('api', function($http) {
  return {
    fetch: function() {
      return $http
        .get('/api/people')
      ;
     }
  };
});


//controller
app.controller('MyController', function($scope, api) {
  $scope.people = [];

  init();

  function init() {
    api
      .fetch()
      .then((response) {
        $scope.people = response.data;
      })
    ;
  }
});
